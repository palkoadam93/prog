SELECT azonosító FROM anyag WHERE mért_egys = 'cm' AND mért_egys = 'm';

SELECT ANYAG.azonosító FROM ANYAG WHERE (((ANYAG.mért_egys)='cm')) OR (((ANYAG.mért_egys)='m'));

SELECT azonosító FROM anyag WHERE mért_egys = 'cm' OR mért_egys = 'm';

SELECT azonosító FROM anyag WHERE mért_egys = 'cm' OR 'm' = mért_egys;

SELECT ANYAG.azonosító FROM ANYAG WHERE (((ANYAG.mért_egys)='cm' OR (ANYAG.mért_egys)='m'));
