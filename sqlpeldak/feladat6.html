6. feladat - Paraméteres lekérdezés

1. Keressük meg, hogy "A" betűvel kezdődő nevű termék van-e az adatbázisban!

SELECT név FROM termék WHERE név LIKE 'A%';

SELECT név FROM termék WHERE Left(név,1) = 'A';

2. Keressük meg, hogy előre nem definiált, a lekérdezés idején megadott karaktersorozattal kezdődő nevű termék van-e az adatbázisban!

SELECT név FROM termék WHERE név LINE [keresett érték] & '%';

sELECT [keresett érték] AS [mit keresek], COUNT(*) AS darabszám, IF(darabszám=0, 'nincs', 'van') AS [van-e] FROM termék WHERE név LIKE [keresett érték] & '%';

SELECT [keresett érték] AS [mit keresek], COUNT(*) AS darabszám, IF(darabszám=0, 'nincs', 'van') AS [van-e] FROM termék WHERE (((termék.név) ALIKE [keresett érték] & '%'));
