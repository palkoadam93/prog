7. feladat - Táblák összekapcsolása, direktszorzat

1. Határozzuk meg, hogy egy adott termék milyen anyagokból épül fel! A termékeknek a nevét és a bennük lévő anyaoknak az azonosítóját írassuk ki!

SELECT termék.*, szerkezet.* FROM termék, szerkezet;

SELECT termék.kód, szerkezet.kód FROM termék, szerkezet;

SELECT termék.kód, szerkezet.kód FROM termék, szerkezet WHERE termék.kód=szerkezet.kód;

SELECT név, termék.kód, szerkezet.kód, azonosító FROM termék, szerkezet WHERE termék.kód=szerkezet.kód;

SELECT név, azonosító FROM termék, szerkezet WHERE termék.kód=szerkezet.kód;

2. A termék nevét és a benne lévő anyagok nevét írjuk ki!

SELECT név, termék.kód, szerkezet.kód, szerkezet.azonosító, anyag.azonosító, neve FROM termék, szerkezet, anyag WHERE termék.kód=szerkezet.kód AND szerkezet.azonosító=anyag.azonosító;

SELECT név, neve FROM termék, szerkezet, anyag WHERE termék.kód=szerkezet.kód AND szerkezet.azonosító=anyag.azonosító;

SELECT név, neve FROM termék, szerkezet, anyag WHERE termék.kód=szerkezet.kód AND szerkezet.azonosító=anyag.azonosító AND termék.kód='AA1';

SELECT neve, név FROM termék, szerkezet, anyag WHERE termék.kód=szerkezet.kód AND szerkezet.azonosító=anyag.azonosító AND anyag.azonosító=112;
